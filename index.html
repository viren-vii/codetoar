void setup(){
    Serial.begin(9600);
    // MOTOR AND ENABLES
    pinMode (RM1, OUTPUT);
    pinMode (RM2, OUTPUT);
    pinMode (LM1, OUTPUT);
    pinMode (LM2, OUTPUT);
    pinMode (ENA, OUTPUT);
    pinMode (ENB, OUTPUT);
    // IR ARRAY
    pinMode (IR1, INPUT);
    pinMode (IR2, INPUT);
    pinMode (IR3, INPUT);
    pinMode (IR4, INPUT);
    pinMode (IR5, INPUT);
    // ULTRASONIC SENSOR
    pinMode(ECHO_PIN,INPUT);
    pinMode(TRIGGER_PIN,OUTPUT);
}
